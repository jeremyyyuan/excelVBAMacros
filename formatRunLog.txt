Sub formatRunLog()
    'Set the current worksheet as the active sheet
    Dim sht As Worksheet
    Set sht = ActiveSheet
    'Expand columns A through Q
    sht.Columns("A:Q").AutoFit
    sht.Range("L2:L200").NumberFormat = "m/dd/yyyy"
    
    sortbyDate
    
    replaceValues
    
    currentRuns
End Sub
Function sortbyDate()
    Dim StRw As Integer, EndRw As Integer
    StRw = 2 ' Starting Row
    EndRw = Range("L65500").End(xlUp).Row
    Rows(StRw & ":" & EndRw).Select
    Selection.Sort Key1:=Range("L2"), Order1:=xlAscending
End Function

Function replaceValues()
    Dim i As Integer, valueToFind1 As String, valueToFind2 As String, valueToFind3 As String, valueToFind4 As String, valueToFind5 As String
    Dim lrow As Integer
    lrow = Cells(Rows.Count, 11).End(xlUp).Row
    valueToFind1 = ""
    valueToFind2 = "|"
    valueToFind3 = "na"
    valueToFind4 = "n/a"
    valueToFind5 = "N/A"
    For i = 1 To lrow
        If (Cells(i, 11).Value = valueToFind1 Or Cells(i, 11).Value = valueToFind2 Or Cells(i, 11).Value = valueToFind3 Or Cells(i, 11).Value = valueToFind4 Or Cells(i, 11).Value = valueToFind5) Then
            Cells(i, 11) = "."
        End If
    Next i
End Function

Function currentRuns()
    Dim i As Integer
    Dim myValue As Variant
    myValue = InputBox("Enter the run Number that you would like to be marked as a current run.")
    Dim lrow As Integer
    lrow = Cells(Rows.Count, 6).End(xlUp).Row
    
    For i = 1 To lrow
        If (CStr(Cells(i, 6).Value) = myValue) Then
            Cells(i, 8) = "Yes"
        End If
    Next i
End Function
